def departments
  unless token?
    render json:  {:status => 0, :error => 'token验证失败'}
    return false
  end

  para = params[:filter]
  _filer = conditioner(para)
  filters = _filer[0]
  condition = _filer[1]
  data = Department.where(condition,filters).limit(filters[:limit]).offset(filters[:offset])

  if data.empty?
    @json = {status00,error:'没有这样的科室'}
  else
    @json = {status11,data:data}
  end

  render json: @json

end
